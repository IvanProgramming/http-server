50Response respondFile(Request request) {
51 std::string filePath = request.getPath().getPath();
52 // Cutting off the leading slash
53 filePath = filePath.substr(1);
54 std::streampos size;
55 std::ifstream file;
56 file.open(filePath, std::ios::binary);
57 std::string extension = getFileExtension(filePath);
58
59 if (file.is_open())
60 {
61 // Get the file size
62 file.seekg(0, std::ios::end);
63 std::streampos fileSize = file.tellg();
64 file.seekg(0, std::ios::beg);
65
66 // Read the file content into a vector
67 std::vector<char> buffer(fileSize);
68 if (!file.read(buffer.data(), fileSize)) {
69 std::cerr << "Failed to read file." << std::endl;
70 return Response("Forbidden", 403, std::map<std::string, std::string>{});
71 }
72
73 std::string mime = MimeByExtension(extension);
74
75 std::map<std::string, std::string> headers = {
76 {"Content-Type", mime},
77 {"Content-Length", std::to_string(size)}
78 };
79
80 // Checking for query parameters named "cache"
81 std::map<std::string, std::string> query = request.getPath().getQuery();
82
83 if (query.find("cache") != query.end() && query["cache"] == "1") {
84 headers["Cache-Control"] = "max-age=3600";
85 }
86
87 return Response(buffer, 200, headers);
88 }
89 else {
90 if (filePath == getFileDirectory(filePath)) {
91 std::cout << "Serving index.html" << std::endl;
92 return RedirectResponse(filePath + "index.html");
93 }
94 std::cout << "File not found" << std::endl;
95 return Response("File not found", 404, std::map<std::string, std::string>{});
96 }
97}
