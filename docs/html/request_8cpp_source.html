177Request CreateRequest(std::string requestString) {
178
179 std::string method;
180 std::string path;
181 std::map<std::string, std::string> headers;
182 std::string body;
183
184 // Parsing request metod and path
185 std::string::size_type pos = requestString.find(' ');
186
187 if (pos == std::string::npos) {
188 throw(ProcessingException("Error parsing request", 400));
189
190 // Invalid request
191 return Request("", "", headers, body);
192 }
193
194 method = requestString.substr(0, pos);
195
196 // Removing method from request string, so we can parse path
197 requestString.erase(0, pos + 1);
198 pos = requestString.find(' ');
199
200 if (pos == std::string::npos) {
201 throw(ProcessingException("Error parsing request", 400));
202
203 // Invalid request
204 return Request("", "", headers, body);
205 }
206
207 path = requestString.substr(0, pos);
208
209 // Removing path from request string, so we can parse headers
210 requestString.erase(0, pos + 1);
211
212 // Any other data is in headers
213 while (true) {
214 if (requestString.find("\r\n") == 0) {
215 // End of headers
216 requestString.erase(0, 2);
217 break;
218 }
219 else {
220 std::string headerName;
221 std::string headerValue;
222
223 headers[headerName] = headerValue;
224
225 int lineEndPos = requestString.find("\r\n");
226
227 if (lineEndPos == std::string::npos) {
228 // End of headers
229 break;
230 }
231
232 // Removing header from request string, so we can parse next header
233 requestString.erase(0, lineEndPos + 2);
234 }
235 }
236
237 // Body is the rest of the request string
238 body = requestString;
239 return Request(method, path, headers, body);
240}

254 for (std::string::const_iterator i = uri.begin(), n = uri.end(); i != n; ++i) {

257 // Keep alphanumeric and other accepted characters intact
